63 {

65  if (size == Dynamic)
66  {
67  A.resize(sp.GetMagnetizationVectors()->rows(), sp.GetMagnetizationVectors()->rows()); // allocate space for dynamic matrices
68  }
69  A.setConstant(0.0); // init A
70 
71  // Get number of CEST pools for matrix size
72  N = sp.GetNumberOfCESTPools();
73 
74  // MT
75  double k_ac = 0.0; // init with 0 for late
76  if (sp.IsMTActive())
77  {
78  double k_ca = sp.GetMTPool()->GetExchangeRateInHz();
79  k_ac = k_ca * sp.GetMTPool()->GetFraction();
80  A(2 * (N + 1), 3 * (N + 1)) = k_ca;
81  A(3 * (N + 1), 2 * (N + 1)) = k_ac;
82  }
83 
84  //WATER
85  double k1a = sp.GetWaterPool()->GetR1() + k_ac;
86  double k2a = sp.GetWaterPool()->GetR2();
87  for (int i = 0; i < N; i++)
88  {
89  double k_ai = sp.GetCESTPool(i)->GetFraction() * sp.GetCESTPool(i)->GetExchangeRateInHz();
90  k1a += k_ai;
91  k2a += k_ai;
92  }
93  A(0, 0) = -k2a;
94  A(1 + N, 1 + N) = -k2a;
95  A(2 + 2 * N, 2 + 2 * N) = -k1a;
96 
97  //CEST POOLS
98  for (int i = 0; i < N; i++)
99  {
100  double k_ia = sp.GetCESTPool(i)->GetExchangeRateInHz();
101  double k_ai = sp.GetCESTPool(i)->GetFraction() * sp.GetCESTPool(i)->GetExchangeRateInHz();
102  double k1i = sp.GetCESTPool(i)->GetR1() + k_ia;
103  double k2i = sp.GetCESTPool(i)->GetR2() + k_ia;
104  ;
105  // 1st submatrix
106  A(0, i + 1) = k_ia;
107  A(i + 1, 0) = k_ai;
108  A(i + 1, i + 1) = -k2i;
109 
110  // 2nd Submatrix
111  A(1 + N, i + 2 + N) = k_ia;
112  A(i + 2 + N, 1 + N) = k_ai;
113  A(i + 2 + N, i + 2 + N) = -k2i;
114 
115  //3rd submatrix
116  A(2 * (N + 1), i + 1 + 2 * (N + 1)) = k_ia;
117  A(i + 1 + 2 * (N + 1), 2 * (N + 1)) = k_ai;
118  A(i + 1 + 2 * (N + 1), i + 1 + 2 * (N + 1)) = -k1i;
119  }
120 
121  // Fill relaxation vector ////
122  if (size == Dynamic) { // alocate space for dynamic matrices
123  C.resize(sp.GetMagnetizationVectors()->rows());
124  }
125 
126  // set entries
127  C.setConstant(0.0);
128  C((N + 1) * 2) = sp.GetWaterPool()->GetFraction() * sp.GetWaterPool()->GetR1(); // water
129  for (int i = 0; i < N; i++) { // vest
130  C((N + 1) * 2 + (i + 1)) = sp.GetCESTPool(i)->GetFraction()*sp.GetCESTPool(i)->GetR1();
131  }
132 
133  if (sp.IsMTActive()) { // set MT related info
134  C(3 * (N + 1)) = sp.GetMTPool()->GetFraction()*sp.GetMTPool()->GetR1();
135  }
136 
137  // set inhomogeneity

139  dw0 = w0 *sp.GetScannerB0Inhom();

155 template<int size> void BlochMcConnellSolver<size>::UpdateBlochMatrix(SimulationParameters &sp, double rfAmplitude, double rfFrequency, double rfPhase)
156 {
157  A(0, 1 + N) = -dw0; // dephasing of water pool
158  A(1 + N, 0) = dw0;
159  // set omega 1
160  double rfAmplitude2pi = rfAmplitude*TWO_PI*sp.GetScannerRelB1();
161  double rfAmplitude2piCosPhi = rfAmplitude2pi * cos(rfPhase);
162  double rfAmplitude2piSinPhi = rfAmplitude2pi * sin(rfPhase);
163  //water
164  A(0, 2 * (N + 1)) = -rfAmplitude2piSinPhi;
165  A(2 * (N + 1), 0) = rfAmplitude2piSinPhi;
166 
167  A(N + 1, 2 * (N + 1)) = rfAmplitude2piCosPhi;
168  A(2 * (N + 1), N + 1) = -rfAmplitude2piCosPhi;
169 
170  //CEST
171  for (int i = 1; i <= N; i++)
172  {
173  A(i, i + 2 * (N + 1)) = -rfAmplitude2piSinPhi;
174  A(i + 2 * (N + 1), i) = rfAmplitude2piSinPhi;
175 
176  A(N + 1 + i, i + 2 * (N + 1)) = rfAmplitude2piCosPhi;
177  A(i + 2 * (N + 1), N + 1 + i) = -rfAmplitude2piCosPhi;
178  }
179 
180  // set off-resonance terms
181  //water
182  double rfFreqOffset2pi = rfFrequency *TWO_PI;
183  A(0, 1 + N) -= rfFreqOffset2pi;
184  A(1 + N, 0) += rfFreqOffset2pi;
185  //cest
186  for (int i = 1; i <= N; i++)
187  {
188  double dwi = sp.GetCESTPool(i - 1)->GetShiftinPPM()*w0 - (rfFreqOffset2pi + dw0);
189  A(i, i + N + 1) = dwi;
190  A(i + N + 1, i) = -dwi;
191  }
192 
193  //set MT term
194  if (sp.IsMTActive()) {
195  A(3 * (N + 1), 3 * (N + 1)) = -sp.GetMTPool()->GetR1() - sp.GetMTPool()->GetExchangeRateInHz() - pow(rfAmplitude2pi, 2)* sp.GetMTPool()->GetMTLineAtCurrentOffset(rfFreqOffset2pi + dw0, w0);
196  }
197 }

204 template<int size> void BlochMcConnellSolver<size>::SolveBlochEquation(VectorNd &M, float& t)
205 {
206  VectorNd AInvT = A.inverse()*C; // helper variable A^-1 * C
207  MatrixNd At = A * t; // helper variable A * t
208  //solve exponential with pade method
209  int infExp; //infinity exponent of the matrix
210  int j;
211  std::frexp(At.template lpNorm<Infinity>(), &infExp); // pade method is only stable if ||A||inf / 2^j <= 0.5
212  j = std::max(0, infExp + 1);
213  At = At * (1.0 / (pow(2, j)));
214  //the algorithm usually starts with D = X = N = Identity and c = 1
215  // since c is alway 0.5 after the first loop, we can start in the second round and init the matrices corresponding to that
216  MatrixNd X(At); // X = A after first loop
217  double c = 0.5; // c = 0.5 after first loop
218  MatrixNd N(At);
219  N.setIdentity();
220  MatrixNd D = N - c * At;
221  N += c * At;
222  bool p = true; // D +- cX is dependent from (-1)^k, fastest way is with changing boolean in the loop
223  double q = numApprox;
224  MatrixNd cX; // helper variable for c * X
225  // run the approximation
226  for (int k = 2; k <= q; k++)
227  {
228  c = c * (q - k + 1) / (k*(2 * q - k + 1));
229  X = At * X;
230  cX = c * X;
231  N = N + cX;
232  if (p)
233  D = D + cX;
234  else
235  D = D - cX;
236  p = !p;
237  }
238  MatrixNd F = D.inverse()*N; // solve D*F = N for F
239  for (int k = 1; k <= j; k++)
240  {
241  F *= F;
242  }
243  M = F * (M + AInvT) - AInvT;
244 }
